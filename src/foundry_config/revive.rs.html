33impl ResolcConfig {
34    /// Returns the `ProjectPathsConfig` sub set of the config.
35    pub fn project_paths(config: &Config) -> ProjectPathsConfig<MultiCompilerLanguage> {
36        let builder = ProjectPathsConfig::builder()
37            .cache(config.cache_path.join(RESOLC_SOLIDITY_FILES_CACHE_FILENAME))
38            .sources(&config.src)
39            .tests(&config.test)
40            .scripts(&config.script)
41            .libs(config.libs.iter())
42            .remappings(config.get_all_remappings())
43            .allowed_path(&config.root)
44            .allowed_paths(&config.libs)
45            .allowed_paths(&config.allow_paths)
46            .include_paths(&config.include_paths)
47            .artifacts(&config.out);
48
49        builder.build_with_root(&config.root)
50    }
51
52    pub fn resolc_settings(config: &Config) -> Result<SolcSettings, SolcError> {
53        config.solc_settings().map(|mut s| {
54            s.extra_settings = ResolcSettings::new(
55                config.resolc.optimizer_mode,
56                config.resolc.heap_size,
57                config.resolc.stack_size,
58                None,
59            );
60            s
61        })
62    }
